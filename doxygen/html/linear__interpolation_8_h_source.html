21template <typename It, typename T>
22AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE Index
23check_bounds(const It begin, const It end, const T& x)
24{
25 const int sz = static_cast<int>(end - begin);
26
27 if ((sz < 2) || (x < *begin)) {
28 return Index{0, Limits::LOWLIM};
29 }
30 if (x > *(begin + (sz - 1))) {
31 return Index{sz - 1, Limits::UPLIM};
32 }
33
34 return Index{0, Limits::VALID};
35}

37template <typename It, typename T>
38AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE Index
39bisection_search(const It begin, const It end, const T& x)
40{
41 auto idx = check_bounds(begin, end, x);
42 if ((idx.lim == Limits::LOWLIM) || (idx.lim == Limits::UPLIM)) {
43 return idx;
44 }
45
46 int il = 0;
47 int ir = end - begin;
48 const T xl = *begin;
49
50 while ((ir - il) > 1) {
51 int mid = (il + ir) >> 1;
52 const T xmid = *(begin + mid);
53
54 if ((x - xmid) * (x - xl) <= 0.0) {
55 ir = mid;
56 } else {
57 il = mid;
58 }
59 }
60 idx.idx = il;
61 return idx;
62}

64template <typename It, typename T>
65AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE Index
66find_index(const It begin, const It end, const T& x, const int hint = 1)
67{
68 auto idx = check_bounds(begin, end, x);
69 if ((idx.lim == Limits::LOWLIM) || (idx.lim == Limits::UPLIM)) {
70 return idx;
71 }
72
73 for (It it = (begin + hint); it < end; ++it) {
74 if (x <= *it) {
75 idx.idx = it - begin - 1;
76 break;
77 }
78 }
79 return idx;
80}

82template <typename C1, typename C2>
83AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
84 typename std::iterator_traits<C2>::value_type
85 linear(
86 const C1 xbegin,
87 const C1 xend,
88 const C2 yinp,
89 const typename std::iterator_traits<C1>::value_type& xout)
90{
91 using DType1 = typename std::iterator_traits<C1>::value_type;
92 const auto idx = bisection_search(xbegin, xend, xout);
93
94 if ((idx.lim == Limits::LOWLIM) || (idx.lim == Limits::UPLIM)) {
95 return yinp[idx.idx];
96 }
97 static constexpr DType1 eps = 1.0e-8;
98 const int j = idx.idx;
99 const auto denom = (xbegin[j + 1] - xbegin[j]);
100 const auto facR = (denom > eps) ? ((xout - xbegin[j]) / denom) : 1.0;
101 const auto facL = static_cast<DType1>(1.0) - facR;
102 return facL * yinp[j] + facR * yinp[j + 1];
103}

112template <typename C1, typename C2>
113inline void
114linear_monotonic(const C1& xinp, const C2& yinp, const C1& xout, C2& yout)
115{
116 static constexpr typename C1::value_type eps = 1.0e-8;
117 AMREX_ASSERT(xinp.size() == yinp.size());
118 AMREX_ASSERT(xout.size() == yout.size());
119
120 int hint = 1;
121 int npts = xout.size();
122 for (int i = 0; i < npts; ++i) {
123 const auto& x = xout[i];
124 const auto idx =
125 find_index(xinp.data(), (xinp.data() + xinp.size()), x, hint);
126
127 if ((idx.lim == Limits::LOWLIM) || (idx.lim == Limits::UPLIM)) {
128 yout[i] = yinp[idx.idx];
129 } else if (idx.lim == Limits::VALID) {
130 int j = idx.idx;
131 const auto denom = (xinp[j + 1] - xinp[j]);
132 const auto facR = (denom > eps) ? ((x - xinp[j]) / denom) : 1.0;
133 const auto facL = static_cast<typename C1::value_type>(1.0) - facR;
134 yout[i] = facL * yinp[j] + facR * yinp[j + 1];
135 }
136 hint = idx.idx + 1;
137 }
138}

156// Angles in yinp should be defined between [-upper_bound, +upper_bound]
157// Use cases would be degrees (upper_bound =360) and radians (upper_bound =2pi)
158template <typename C1, typename C2>
159AMREX_GPU_HOST_DEVICE AMREX_FORCE_INLINE
160 typename std::iterator_traits<C2>::value_type
161 linear_angle(
162 const C1 xbegin,
163 const C1 xend,
164 const C2 yinp,
165 const typename std::iterator_traits<C1>::value_type& xout,
166 const typename std::iterator_traits<C1>::value_type& upper_bound)
167{
168 using DType1 = typename std::iterator_traits<C1>::value_type;
169 const auto idx = bisection_search(xbegin, xend, xout);
170
171 if ((idx.lim == Limits::LOWLIM) || (idx.lim == Limits::UPLIM)) {
172 return yinp[idx.idx];
173 }
174 static constexpr DType1 eps = 1.0e-8;
175 const int j = idx.idx;
176 const auto denom = (xbegin[j + 1] - xbegin[j]);
177 /* https://math.stackexchange.com/questions/2144234/interpolating-between-2-angles
178 */
179 const auto R = (denom > eps) ? ((xout - xbegin[j]) / denom) : 0.0;
180 const auto ub = upper_bound;
181 const auto hb = 0.5 * upper_bound;
182 const auto ohb = 1.5 * upper_bound;
183 return yinp[j] +
184 (std::fmod((std::fmod(yinp[j + 1] - yinp[j], ub) + ohb), ub) - hb) *
185 R;
186}
